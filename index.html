food_count <- latest_data %>%
  count(food_type)

latest_data %>%
  group_by(food_type) %>%
  summarise(avg_queue = mean(queue_length, na.rm = TRUE)) %>%
  ggplot(aes(x = food_type, y = avg_queue)) +
  geom_col(fill = "darkorange") +
  labs(title = "Average Queue Length by Food Type",
       x = "Food Type", y = "Average Queue Length") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

queue_summary <- latest_data %>%
  group_by(food_type) %>%
  summarise(avg_queue = mean(queue_length, na.rm = TRUE))

paste("The average queue length for", queue_summary$food_type[1], "is", round(queue_summary$avg_queue[1], 2), 
      "and for", queue_summary$food_type[2], "is", round(queue_summary$avg_queue[2], 2))
## [1] "The average queue length for Drinks or desserts is 19.8 and for Noodle-based dishes (e.g., ramen, stir-fried noodles) is 24.14"
